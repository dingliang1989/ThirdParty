cmake_minimum_required(VERSION 3.5)

project(cityhash)

file(GLOB_RECURSE CITYHASH_SOURCE_FILES "${CMAKE_CURRENT_SOURCE_DIR}/*.cc" "${CMAKE_CURRENT_SOURCE_DIR}/*.h" "${CMAKE_CURRENT_SOURCE_DIR}/*.h.in")

source_group("src" FILES ${CITYHASH_SOURCE_FILES})

set(CITYHASH_PROJECT_FILES)
list(APPEND CITYHASH_PROJECT_FILES ${CITYHASH_HEADER_FILES})
list(APPEND CITYHASH_PROJECT_FILES ${CITYHASH_SOURCE_FILES})

add_library(${PROJECT_NAME} STATIC ${CITYHASH_PROJECT_FILES})

target_include_directories(${PROJECT_NAME} PUBLIC ${CMAKE_CURRENT_SOURCE_DIR} ${CMAKE_CURRENT_SOURCE_DIR}/src)

add_library(lib::${PROJECT_NAME} ALIAS ${PROJECT_NAME})
